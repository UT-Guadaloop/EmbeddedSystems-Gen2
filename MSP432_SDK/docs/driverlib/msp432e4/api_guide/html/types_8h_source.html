42 // Macros for hardware access, both direct and via the bit-band region.
43 //
44 //*****************************************************************************
45 #define HWREG(x) \
46  (*((volatile uint32_t *)(x)))
47 #define HWREGH(x) \
48  (*((volatile uint16_t *)(x)))
49 #define HWREGB(x) \
50  (*((volatile uint8_t *)(x)))
51 #define HWREGBITW(x, b) \
52  HWREG(((uint32_t)(x) & 0xF0000000) | 0x02000000 | \
53  (((uint32_t)(x) & 0x000FFFFF) << 5) | ((b) << 2))
54 #define HWREGBITH(x, b) \
55  HWREGH(((uint32_t)(x) & 0xF0000000) | 0x02000000 | \
56  (((uint32_t)(x) & 0x000FFFFF) << 5) | ((b) << 2))
57 #define HWREGBITB(x, b) \
58  HWREGB(((uint32_t)(x) & 0xF0000000) | 0x02000000 | \
59  (((uint32_t)(x) & 0x000FFFFF) << 5) | ((b) << 2))
